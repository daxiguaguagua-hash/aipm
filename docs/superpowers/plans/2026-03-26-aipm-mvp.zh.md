# aipm - AI 编程栈管理器 MVP 实现计划

> **对于智能体工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 按任务逐步实现本计划。步骤使用复选框 (`- [ ]`) 语法跟踪。

**目标：** 构建一个可用的 aipm 最小原型，支持 stack.yaml 声明式配置、依赖安装到全局缓存、通过一条 `aipm use` 命令导出配置到 Claude Code / OpenClaw / OpenCode 三个目标平台。

**架构：** 核心层级：CLI → 协调器 → 解析器/安装器/导出器 → 各目标平台适配器。全局缓存位于 `~/.aipm/cache/`，存储已安装的 skills/agents/MCPs。声明式 `stack.yaml` 是唯一真相源。适配器模式隔离平台特定的转换逻辑。

**技术栈：** TypeScript + Node.js，使用 `js-yaml` 解析 YAML，Commander.js 处理 CLI，`simple-git` 处理 git 操作。

```typescript
  if (!fs.existsSync(dir)) {
```

```typescript
import fs from 'fs';
import path from 'path';
import { InstalledComponent } from './types';
import { ensureDir, getGlobalCacheDir, logInfo, logSuccess } from './utils';

// 缓存管理器 - 管理全局缓存中的已安装组件
export class CacheManager {
  private cacheDir: string;

  constructor() {
    this.cacheDir = getGlobalCacheDir();
    ensureDir(this.cacheDir);
  }

  // 获取组件缓存路径
  getCachePath(id: string): string {
    return path.join(this.cacheDir, id);
  }

  // 检查组件是否已安装
  isInstalled(id: string): boolean {
    return fs.existsSync(this.getCachePath(id));
  }

  // 获取已安装组件的元数据
  getInstalledMetadata(id: string): InstalledComponent | null {
    const metaPath = path.join(this.getCachePath(id), '.aipm-meta.json');
    if (!fs.existsSync(metaPath)) {
      return null;
    }
    try {
      return JSON.parse(fs.readFileSync(metaPath, 'utf8'));
    } catch {
      return null;
    }
  }

  // 保存元数据
  saveMetadata(meta: InstalledComponent): void {
    const metaPath = path.join(meta.path ? path.dirname(meta.path) : this.getCachePath(meta.id), '.aipm-meta.json');
    const fullPath = path.join(this.getCachePath(meta.id), '.aipm-meta.json');
    fs.writeFileSync(fullPath, JSON.stringify(meta, null, 2));
    logSuccess(`保存 ${meta.id} 元数据完成`);
  }

  // 列出所有已安装组件
  listInstalled(): InstalledComponent[] {
    ensureDir(this.cacheDir);
    const items = fs.readdirSync(this.cacheDir);
    const result: InstalledComponent[] = [];
    for (const item of items) {
      const meta = this.getInstalledMetadata(item);
      if (meta) {
        result.push(meta);
      }
    }
    return result;
  }
}
```

```typescript
import yaml from 'js-yaml';
import { StackConfig } from './types';
import { logError } from './utils';

// 解析 YAML 内容为 StackConfig
export function parseStackConfig(content: string): StackConfig {
  try {
    const config = yaml.load(content) as any;

    // 基本验证
    if (!config.project || typeof config.project !== 'string') {
      throw new Error('缺少或无效的 "project" 字段');
    }

    if (!config.targets || typeof config.targets !== 'object') {
      throw new Error('缺少或无效的 "targets" 字段');
    }

    const stackConfig: StackConfig = {
      project: config.project,
      skills: config.skills || [],
      agents: config.agents || [],
      mcps: config.mcps || [],
      targets: config.targets,
    };

    return stackConfig;
  } catch (error) {
    if (error instanceof Error) {
      logError(`YAML 解析错误: ${error.message}`);
      throw error;
    }
    throw new Error('未知解析错误');
  }
}

// 从文件加载并解析
export function loadStackConfigFromFile(filePath: string): StackConfig {
  const fs = require('fs');
  const content = fs.readFileSync(filePath, 'utf8');
  return parseStackConfig(content);
}
```

```typescript
import simpleGit from 'simple-git';
import path from 'path';
import { CacheManager } from './cache';
import { Skill, Agent, MCP, InstalledComponent } from './types';
import { ensureDir, extractGitRepoName, logInfo, logSuccess, logError } from './utils';

// 安装器 - 将组件安装到全局缓存
export class Installer {
  constructor(private cacheManager: CacheManager) {}

  // 检查是否已安装
  isAlreadyInstalled(id: string): boolean {
    return this.cacheManager.isInstalled(id);
  }

  // 安装一个技能
  async installSkill(skill: Skill): Promise<InstalledComponent> {
    if (this.cacheManager.isInstalled(skill.id)) {
      logInfo(`技能 ${skill.id} 已安装`);
      const meta = this.cacheManager.getInstalledMetadata(skill.id);
      if (meta) return meta;
    }

    const installPath = this.cacheManager.getCachePath(skill.id);
    ensureDir(path.dirname(installPath));

    // 解析源地址：支持 git+https, github:owner/repo, 本地路径
    let repoUrl = skill.source;
    if (repoUrl.startsWith('github:')) {
      repoUrl = `https://github.com/${repoUrl.slice(7)}.git`;
    }
    if (repoUrl.startsWith('git+')) {
      repoUrl = repoUrl.slice(4);
    }

    // 检查是否是本地路径
    if (repoUrl.startsWith('.') || repoUrl.startsWith('/')) {
      // 本地技能 - 只记录元数据
      const meta: InstalledComponent = {
        id: skill.id,
        source: skill.source,
        version: skill.version || 'local',
        path: repoUrl,
      };
      this.cacheManager.saveMetadata(meta);
      logSuccess(`安装本地技能 ${skill.id} 完成`);
      return meta;
    }

    // Clone git 仓库
    logInfo(`克隆 ${repoUrl}...`);
    try {
      await simpleGit().clone(repoUrl, installPath);
      // 获取最新 commit hash 作为版本
      const git = simpleGit(installPath);
      const hash = await git.revparse(['HEAD']);
      const version = hash.trim().slice(0, 7);

      const meta: InstalledComponent = {
        id: skill.id,
        source: skill.source,
        version,
        path: installPath,
      };
      this.cacheManager.saveMetadata(meta);
      logSuccess(`安装技能 ${skill.id}@${version} 完成`);
      return meta;
    } catch (error) {
      logError(`安装技能 ${skill.id} 失败: ${error}`);
      throw error;
    }
  }

  // 安装一个 Agent
  async installAgent(agent: Agent, source: string): Promise<InstalledComponent> {
    // Agent 也可以从 git 安装，流程和 skill 类似
    return this.installSkill({
      id: agent.id,
      source,
      entry: agent.system || './system.md',
    });
  }

  // 安装一个 MCP
  async installMcp(mcp: MCP, source: string): Promise<InstalledComponent> {
    // 内联定义的 MCP 不需要安装
    if (!source) {
      return {
        id: mcp.id,
        source: 'inline',
        version: 'inline',
        path: '',
      };
    }
    return this.installSkill({
      id: mcp.id,
      source,
      entry: './index.js',
    });
  }

  // 批量安装所有组件
  async installAll(
    skills: Skill[],
    agents: Agent[],
    mcps: MCP[]
  ): Promise<{
    skills: InstalledComponent[];
    agents: InstalledComponent[];
    mcps: InstalledComponent[];
  }> {
    const installedSkills = await Promise.all(
      skills.map(skill => this.installSkill(skill))
    );
    const installedAgents = await Promise.all(
      agents.map(agent => this.installAgent(agent, agent.id))
    );
    const installedMcps = await Promise.all(
      mcps.map(mcp => this.installMcp(mcp, mcp.id))
    );
    return {
      skills: installedSkills,
      agents: installedAgents,
      mcps: installedMcps,
    };
  }
}
```

```typescript
import fs from 'fs';
import path from 'path';
import { Adapter, StackConfig, InstalledComponent } from '../../types';
import { ensureDir, logSuccess } from '../../utils';

// Claude Code 平台适配器
export class ClaudeCodeAdapter implements Adapter {
  async exportConfig(
    stack: StackConfig,
    installed: { skills: InstalledComponent[]; agents: InstalledComponent[]; mcps: InstalledComponent[] },
    outputDir: string
  ): Promise<void> {
    ensureDir(outputDir);

    // Claude Code 使用 .claude/ 目录
    const claudeDir = path.join(outputDir, '.claude');
    ensureDir(claudeDir);

    // 导出 skills 配置：链接到缓存中的入口文件
    const skillsConfig: any = {};
    for (const skill of installed.skills) {
      const original = stack.skills?.find(s => s.id === skill.id);
      if (original) {
        const entryPath = path.join(skill.path, original.entry);
        skillsConfig[skill.id] = {
          path: entryPath,
          version: skill.version,
        };
      }
    }

    // 导出 MCP 配置
    const mcpConfig: any = {
      mcpServers: {},
    };
    for (const mcp of stack.mcps || []) {
      mcpConfig.mcpServers[mcp.id] = {
        command: mcp.command,
        args: mcp.args || [],
        env: mcp.env || {},
      };
    }

    // 写入文件
    if (Object.keys(skillsConfig).length > 0) {
      fs.writeFileSync(
        path.join(claudeDir, 'skills.json'),
        JSON.stringify(skillsConfig, null, 2)
      );
    }
    if (Object.keys(mcpConfig.mcpServers).length > 0) {
      fs.writeFileSync(
        path.join(claudeDir, 'mcp.json'),
        JSON.stringify(mcpConfig, null, 2)
      );
    }

    logSuccess(`导出 Claude Code 配置到 ${outputDir}`);
  }
}
```

```typescript
import path from 'path';
import { getAdapter } from './adapters';
import { StackConfig, TargetPlatform, InstalledComponent } from './types';
import { getLocalExportsDir, ensureDir } from './utils';

// 导出器 - 将配置导出到目标平台
export class Exporter {
  async exportToTarget(
    stack: StackConfig,
    target: TargetPlatform,
    installed: { skills: InstalledComponent[]; agents: InstalledComponent[]; mcps: InstalledComponent[] },
    outputDir?: string
  ): Promise<void> {
    const actualOutputDir = outputDir || path.join(getLocalExportsDir(), target);
    ensureDir(actualOutputDir);

    const adapter = getAdapter(target);
    await adapter.exportConfig(stack, installed, actualOutputDir);
  }

  async exportAllTargets(
    stack: StackConfig,
    installed: { skills: InstalledComponent[]; agents: InstalledComponent[]; mcps: InstalledComponent[] }
  ): Promise<void> {
    for (const target of Object.keys(stack.targets) as TargetPlatform[]) {
      await this.exportToTarget(stack, target, installed);
    }
  }
}
```

- [ ] **步骤 2：实现协调器**

```typescript
import fs from 'fs';
import path from 'path';
import { StackConfig } from './types';
import { CacheManager } from './cache';
import { Installer } from './installer';
import { Exporter } from './exporter';
import { loadStackConfigFromFile } from './parser';
import { getLocalAiDir, logInfo, logSuccess } from './utils';

// 顶层协调器 - 统筹整个工作流
export class Orchestrator {
  private cacheManager: CacheManager;
  private installer: Installer;
  private exporter: Exporter;

  constructor() {
    this.cacheManager = new CacheManager();
    this.installer = new Installer(this.cacheManager);
    this.exporter = new Exporter();
  }

  // 加载本地 stack.yaml
  loadStackConfig(): StackConfig {
    const stackPath = path.join(getLocalAiDir(), 'stack.yaml');
    if (!fs.existsSync(stackPath)) {
      throw new Error(`在 ${stackPath} 未找到 stack.yaml。请先运行 'aipm init'。`);
    }
    return loadStackConfigFromFile(stackPath);
  }

  // 安装所有依赖
  async install(stack: StackConfig): Promise<{
    skills: InstalledComponent[];
    agents: InstalledComponent[];
    mcps: InstalledComponent[];
  }> {
    logInfo('正在安装依赖...');
    const result = await this.installer.installAll(
      stack.skills || [],
      stack.agents || [],
      stack.mcps || []
    );
    logSuccess('安装完成');
    return result;
  }

  // 导出配置到目标平台
  async export(stack: StackConfig, target: string, installed: any): Promise<void> {
    logInfo(`正在导出到 ${target}...`);
    await this.exporter.exportToTarget(stack, target as any, installed);
    logSuccess(`导出到 ${target} 完成`);
  }

  // use 命令：一键安装 + 导出
  async use(stack: StackConfig, target: string): Promise<void> {
    const installed = await this.install(stack);
    await this.export(stack, target, installed);
    logSuccess(`就绪！现在你可以在 ${target} 中使用你配置的 AI 栈了。`);
  }
}
```

```typescript
#!/usr/bin/env node

import { Command } from 'commander';
import chalk from 'chalk';
import fs from 'fs';
import path from 'path';
import { Orchestrator } from './orchestrator';
import { getLocalAiDir, ensureDir, logError, logSuccess } from './utils';

const program = new Command();
const orchestrator = new Orchestrator();

program
  .name('aipm')
  .description('AI Coding Stack Manager - AI 编程栈管理器')
  .version('0.1.0');

// init 命令：初始化新项目
program
  .command('init')
  .description('初始化新的 AI 栈项目')
  .action(() => {
    const aiDir = getLocalAiDir();
    ensureDir(aiDir);

    const stackPath = path.join(aiDir, 'stack.yaml');
    if (fs.existsSync(stackPath)) {
      logError('stack.yaml 已存在');
      process.exit(1);
    }

    const template = `# aipm AI 栈配置
project: my-ai-stack

# 你要使用的技能
skills:
#  - id: code-review
#    source: github:anthonyclays/code-review-skill
#    entry: ./main.md

# 你要使用的 Agents
agents:
#  - id: planner
#    model: claude-sonnet
#    skills:
#      - code-review

# 你要使用的 MCP 服务器
mcps:
#  - id: filesystem
#    command: npx
#    args: ["-y", "@modelcontextprotocol/server-filesystem", "."]

# 要导出的目标平台
targets:
  claude-code:
    agents: []
    skills: []
    mcps: []
  openclaw:
    agents: []
    skills: []
    mcps: []
  opencode:
    skills: []
    mcps: []
`;

    fs.writeFileSync(stackPath, template);
    ensureDir(path.join(aiDir, 'exports'));
    logSuccess('在 .ai/ 初始化 aipm 项目完成');
  });

// install 命令：安装所有依赖到全局缓存
program
  .command('install')
  .description('安装所有依赖到全局缓存')
  .action(async () => {
    try {
      const stack = orchestrator.loadStackConfig();
      await orchestrator.install(stack);
    } catch (error) {
      if (error instanceof Error) {
        logError(error.message);
      }
      process.exit(1);
    }
  });

// export 命令：导出配置到目标平台
program
  .command('export <target>')
  .description('导出配置到目标平台')
  .action(async (target) => {
    try {
      const stack = orchestrator.loadStackConfig();
      const installed = await orchestrator.install(stack);
      await orchestrator.export(stack, target, installed);
    } catch (error) {
      if (error instanceof Error) {
        logError(error.message);
      }
      process.exit(1);
    }
  });

// use 命令：切换到目标平台 - 一键完成安装和导出
program
  .command('use <target>')
  .description('切换到目标平台 - 一条命令完成安装和导出')
  .action(async (target) => {
    try {
      const stack = orchestrator.loadStackConfig();
      await orchestrator.use(stack, target);
    } catch (error) {
      if (error instanceof Error) {
        logError(error.message);
      }
      process.exit(1);
    }
  });

program.parseAsync().catch(error => {
  logError(error.message);
  process.exit(1);
});
```
