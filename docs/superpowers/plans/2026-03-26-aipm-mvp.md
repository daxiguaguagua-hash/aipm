# aipm - AI Coding Stack Manager MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a minimal viable prototype of aipm that supports declarative configuration in stack.yaml, dependency installation to global cache, and exporting configuration to Claude Code / OpenClaw / OpenCode targets with a single `aipm use` command.

**Architecture:** Core layers: CLI → Orchestrator → Parser/Installer/Exporter → Adapters for each target platform. Global cache at `~/.aipm/cache/` stores installed skills/agents/MCPs. Declarative `stack.yaml` is the source of truth. Adapter pattern isolates platform-specific conversion logic.

**Tech Stack:** TypeScript + Node.js, YAML parsing via `js-yaml`, Commander.js for CLI, simple git operations via `simple-git`.

```typescript
import fs from 'fs';
import path from 'path';
import { InstalledComponent } from './types';
import { ensureDir, getGlobalCacheDir, logInfo, logSuccess } from './utils';

export class CacheManager {
  private cacheDir: string;

  constructor() {
    this.cacheDir = getGlobalCacheDir();
    ensureDir(this.cacheDir);
  }

  getCachePath(id: string): string {
    return path.join(this.cacheDir, id);
  }

  isInstalled(id: string): boolean {
    return fs.existsSync(this.getCachePath(id));
  }

  getInstalledMetadata(id: string): InstalledComponent | null {
    const metaPath = path.join(this.getCachePath(id), '.aipm-meta.json');
    if (!fs.existsSync(metaPath)) {
      return null;
    }
    try {
      return JSON.parse(fs.readFileSync(metaPath, 'utf8'));
    } catch {
      return null;
    }
  }

  saveMetadata(meta: InstalledComponent): void {
    const metaPath = path.join(this.getCachePath(meta.id), '.aipm-meta.json');
    fs.writeFileSync(metaPath, JSON.stringify(meta, null, 2));
    logSuccess(`Saved metadata for ${meta.id}`);
  }

  listInstalled(): InstalledComponent[] {
    ensureDir(this.cacheDir);
    const items = fs.readdirSync(this.cacheDir);
    const result: InstalledComponent[] = [];
    for (const item of items) {
      const meta = this.getInstalledMetadata(item);
      if (meta) {
        result.push(meta);
      }
    }
    return result;
  }
}
```

```typescript
import yaml from 'js-yaml';
import { StackConfig } from './types';
import { logError } from './utils';

export function parseStackConfig(content: string): StackConfig {
  try {
    const config = yaml.load(content) as any;

    // Basic validation
    if (!config.project || typeof config.project !== 'string') {
      throw new Error('Missing or invalid "project" field');
    }

    if (!config.targets || typeof config.targets !== 'object') {
      throw new Error('Missing or invalid "targets" field');
    }

    const stackConfig: StackConfig = {
      project: config.project,
      skills: config.skills || [],
      agents: config.agents || [],
      mcps: config.mcps || [],
      targets: config.targets,
    };

    return stackConfig;
  } catch (error) {
    if (error instanceof Error) {
      logError(`YAML parse error: ${error.message}`);
      throw error;
    }
    throw new Error('Unknown parse error');
  }
}

export function loadStackConfigFromFile(filePath: string): StackConfig {
  const fs = require('fs');
  const content = fs.readFileSync(filePath, 'utf8');
  return parseStackConfig(content);
}
```

```typescript
import simpleGit from 'simple-git';
import path from 'path';
import { CacheManager } from './cache';
import { Skill, Agent, MCP, InstalledComponent } from './types';
import { ensureDir, extractGitRepoName, logInfo, logSuccess, logError } from './utils';

export class Installer {
  constructor(private cacheManager: CacheManager) {}

  isAlreadyInstalled(id: string): boolean {
    return this.cacheManager.isInstalled(id);
  }

  async installSkill(skill: Skill): Promise<InstalledComponent> {
    if (this.cacheManager.isInstalled(skill.id)) {
      logInfo(`Skill ${skill.id} already installed`);
      const meta = this.cacheManager.getInstalledMetadata(skill.id);
      if (meta) return meta;
    }

    const installPath = this.cacheManager.getCachePath(skill.id);
    ensureDir(path.dirname(installPath));

    // Parse source: support git+https, github:owner/repo, local path
    let repoUrl = skill.source;
    if (repoUrl.startsWith('github:')) {
      repoUrl = `https://github.com/${repoUrl.slice(7)}.git`;
    }
    if (repoUrl.startsWith('git+')) {
      repoUrl = repoUrl.slice(4);
    }

    // Check if it's a local path
    if (repoUrl.startsWith('.') || repoUrl.startsWith('/')) {
      // Local skill - just symlink/copy metadata
      const meta: InstalledComponent = {
        id: skill.id,
        source: skill.source,
        version: skill.version || 'local',
        path: repoUrl,
      };
      this.cacheManager.saveMetadata(meta);
      logSuccess(`Installed local skill ${skill.id}`);
      return meta;
    }

    // Clone git repo
    logInfo(`Cloning ${repoUrl}...`);
    try {
      await simpleGit().clone(repoUrl, installPath);
      // Get latest commit hash as version
      const git = simpleGit(installPath);
      const hash = await git.revparse(['HEAD']);
      const version = hash.trim().slice(0, 7);

      const meta: InstalledComponent = {
        id: skill.id,
        source: skill.source,
        version,
        path: installPath,
      };
      this.cacheManager.saveMetadata(meta);
      logSuccess(`Installed skill ${skill.id}@${version}`);
      return meta;
    } catch (error) {
      logError(`Failed to install skill ${skill.id}: ${error}`);
      throw error;
    }
  }

  async installAgent(agent: Agent, source: string): Promise<InstalledComponent> {
    // Similar to skill installation for now
    // Agents can come from git as well
    return this.installSkill({
      id: agent.id,
      source,
      entry: agent.system || './system.md',
    });
  }

  async installMcp(mcp: MCP, source: string): Promise<InstalledComponent> {
    // MCPs defined inline don't need installation
    if (!source) {
      return {
        id: mcp.id,
        source: 'inline',
        version: 'inline',
        path: '',
      };
    }
    return this.installSkill({
      id: mcp.id,
      source,
      entry: './index.js',
    });
  }

  async installAll(
    skills: Skill[],
    agents: Agent[],
    mcps: MCP[]
  ): Promise<{
    skills: InstalledComponent[];
    agents: InstalledComponent[];
    mcps: InstalledComponent[];
  }> {
    const installedSkills = await Promise.all(
      skills.map(skill => this.installSkill(skill))
    );
    const installedAgents = await Promise.all(
      agents.map(agent => this.installAgent(agent, agent.id))
    );
    const installedMcps = await Promise.all(
      mcps.map(mcp => this.installMcp(mcp, mcp.id))
    );
    return {
      skills: installedSkills,
      agents: installedAgents,
      mcps: installedMcps,
    };
  }
}
```

```typescript
import fs from 'fs';
import path from 'path';
import { Adapter, StackConfig, InstalledComponent } from '../types';
import { ensureDir, logSuccess } from '../utils';

export class ClaudeCodeAdapter implements Adapter {
  async exportConfig(
    stack: StackConfig,
    installed: { skills: InstalledComponent[]; agents: InstalledComponent[]; mcps: InstalledComponent[] },
    outputDir: string
  ): Promise<void> {
    ensureDir(outputDir);

    // Claude Code uses .claude/ directory
    const claudeDir = path.join(outputDir, '.claude');
    ensureDir(claudeDir);

    // Export skills: link to cached entry files
    const skillsConfig: any = {};
    for (const skill of installed.skills) {
      // Find the original skill entry from stack config
      const original = stack.skills?.find(s => s.id === skill.id);
      if (original) {
        const entryPath = path.join(skill.path, original.entry);
        skillsConfig[skill.id] = {
          path: entryPath,
          version: skill.version,
        };
      }
    }

    // Export MCPs
    const mcpConfig: any = {
      mcpServers: {},
    };
    for (const mcp of stack.mcps || []) {
      mcpConfig.mcpServers[mcp.id] = {
        command: mcp.command,
        args: mcp.args || [],
        env: mcp.env || {},
      };
    }

    // Write files
    if (Object.keys(skillsConfig).length > 0) {
      fs.writeFileSync(
        path.join(claudeDir, 'skills.json'),
        JSON.stringify(skillsConfig, null, 2)
      );
    }
    if (Object.keys(mcpConfig.mcpServers).length > 0) {
      fs.writeFileSync(
        path.join(claudeDir, 'mcp.json'),
        JSON.stringify(mcpConfig, null, 2)
      );
    }

    logSuccess(`Exported Claude Code configuration to ${outputDir}`);
  }
}
```

```typescript
import fs from 'fs';
import path from 'path';
import { StackConfig } from './types';
import { CacheManager } from './cache';
import { Installer } from './installer';
import { Exporter } from './exporter';
import { loadStackConfigFromFile } from './parser';
import { getLocalAiDir, logInfo, logSuccess } from './utils';

export class Orchestrator {
  private cacheManager: CacheManager;
  private installer: Installer;
  private exporter: Exporter;

  constructor() {
    this.cacheManager = new CacheManager();
    this.installer = new Installer(this.cacheManager);
    this.exporter = new Exporter();
  }

  loadStackConfig(): StackConfig {
    const stackPath = path.join(getLocalAiDir(), 'stack.yaml');
    if (!fs.existsSync(stackPath)) {
      throw new Error(`No stack.yaml found at ${stackPath}. Run 'aipm init' first.`);
    }
    return loadStackConfigFromFile(stackPath);
  }

  async install(stack: StackConfig): Promise<{
    skills: InstalledComponent[];
    agents: InstalledComponent[];
    mcps: InstalledComponent[];
  }> {
    logInfo('Installing dependencies...');
    const result = await this.installer.installAll(
      stack.skills || [],
      stack.agents || [],
      stack.mcps || []
    );
    logSuccess('Installation complete');
    return result;
  }

  async export(stack: StackConfig, target: string, installed: any): Promise<void> {
    logInfo(`Exporting to ${target}...`);
    await this.exporter.exportToTarget(stack, target as any, installed);
    logSuccess(`Exported to ${target}`);
  }

  async use(stack: StackConfig, target: string): Promise<void> {
    const installed = await this.install(stack);
    await this.export(stack, target, installed);
    logSuccess(`Ready! Now you can use ${target} with your configured AI stack.`);
  }
}
```
